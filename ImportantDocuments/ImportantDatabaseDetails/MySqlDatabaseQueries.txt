-- Create database if it does not exist
CREATE DATABASE IF NOT EXISTS secure_vault;

-- Select the database
USE secure_vault;

-- Create users table
CREATE TABLE users (
    id INTEGER PRIMARY KEY,
    username TEXT NOT NULL,
    password TEXT NOT NULL,
    role TEXT NOT NULL
);
