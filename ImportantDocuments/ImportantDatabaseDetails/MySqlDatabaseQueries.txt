-- ============================================================
-- Run this ONE TIME in your MySQL secure_vault database
-- BEFORE restarting the Spring Boot application.
--
-- This removes the plain-text password seed records so that
-- DataInitializer.java can re-insert them with BCrypt hashes
-- on the next application startup.
-- ============================================================

USE secure_vault;

-- Remove existing plain-text seed records
DELETE FROM users WHERE id IN (1, 2, 3);

-- Verify table is clear (DataInitializer will re-populate on next startup)
SELECT COUNT(*) AS remaining_rows FROM users;


-- Create database if it does not exist
CREATE DATABASE IF NOT EXISTS secure_vault;

-- Select the database
USE secure_vault;

-- Create users table
CREATE TABLE users (
    id INTEGER PRIMARY KEY,
    username TEXT NOT NULL,
    password TEXT NOT NULL,
    role TEXT NOT NULL
);

-- Insert sample records
INSERT INTO users (id, username, password, role) VALUES
    (1, 'admin_user', 'admin123', 'ADMIN'),
    (2, 'normal_user', 'user123', 'USER'),
    (3, 'manager_user', 'manager123', 'MANAGER');

TRUNCATE TABLE users;


net stop MySQL80

net start MySQL80
